/* global AFRAME */
 
var THREE = AFRAME.THREE;
 
/**
 * A-Frame component that loads a GeoJSON asset.
 */
AFRAME.registerComponent('geojson-loader', {
  schema: {
    src: {
      type: 'asset'
    },
    isCCW: {
      type: 'boolean',
      default: false
    }
  },
 
  /**
   * Called once when component is attached. Generally for initial setup.
   */
  init: function () {
    this.loader = new THREE.FileLoader();
  },
 
  /**
   * Called when component is attached and when component data changes.
   * Generally modifies the entity based on the data.
   */
  update: function (oldData) {
    var src = this.data.src;
    if (src && src !== oldData.src) {
      this.loader.load(src, this.onSrcLoaded.bind(this));
    }
  },
 
  onSrcLoaded: function (text) {
    var geoJson = JSON.parse(text);
    var details = {
      geoJson: geoJson,
      isCCW: this.data.isCCW
    };
    this.el.emit('geojson-loaded', details);
  },
 
  /**
   * Called when a component is removed (e.g., via removeAttribute).
   * Generally undoes all modifications to the entity.
   */
  remove: function () {
  }
});
